# 3.1 Analyze Customers
def analyze_customers():
    print("Customers by Region:")
    region_counts = customers['Region'].value_counts()
    print(region_counts)
    
    plt.figure(figsize=(8, 5))
    sns.countplot(data=customers, x='Region', order=region_counts.index, color='skyblue')
    plt.title('Customer Distribution by Region')
    plt.xticks(rotation=45)
    plt.show()

    # Signup trend over time
    customers['SignupDate'] = pd.to_datetime(customers['SignupDate'])
    signup_trend = customers['SignupDate'].dt.year.value_counts().sort_index()
    print("Signup Trend by Year:\n", signup_trend)

    plt.figure(figsize=(8, 5))
    signup_trend.plot(kind='bar', color='teal')
    plt.title('Customer Signup Trend by Year')
    plt.xlabel('Year')
    plt.ylabel('Number of Signups')
    plt.show()

def analyze_transactions():
    print("Transaction Dates Range:")
    print(transactions['TransactionDate'].min(), "to", transactions['TransactionDate'].max())

    print("Top Products by Quantity Sold:")
    top_products = transactions.groupby('ProductID')['Quantity'].sum().sort_values(ascending=False).head(10)
    print(top_products)

    plt.figure(figsize=(10, 6))
    top_products.plot(kind='bar', color='teal')
    plt.title('Top 10 Products by Quantity Sold')
    plt.ylabel('Total Quantity')
    plt.show()

    # Monthly sales trend
    transactions['TransactionDate'] = pd.to_datetime(transactions['TransactionDate'])
    monthly_sales = transactions.groupby(transactions['TransactionDate'].dt.to_period('M'))['TotalValue'].sum()

    print("Monthly Sales Trend:\n", monthly_sales)

    plt.figure(figsize=(12, 6))
    monthly_sales.plot()
    plt.title('Monthly Sales Trend')
    plt.xlabel('Month')
    plt.ylabel('Total Sales (USD)')
    plt.grid()
    plt.show()

# 3.5 Merged Dataset Analysis
def merged_analysis():
    merged_data = transactions.merge(customers, on='CustomerID').merge(products, on='ProductID')
    
    print("Sales by Region:")
    sales_by_region = merged_data.groupby('Region')['TotalValue'].sum().sort_values(ascending=False)
    print(sales_by_region)

    plt.figure(figsize=(10, 6))
    sales_by_region.plot(kind='bar', color='orange')
    plt.title('Total Sales by Region')
    plt.ylabel('Total Sales (USD)')
    plt.show()

    print("Top Categories by Sales:")
    sales_by_category = merged_data.groupby('Category')['TotalValue'].sum().sort_values(ascending=False)
    print(sales_by_category)

    plt.figure(figsize=(10, 6))
    sales_by_category.plot(kind='bar', color='green')
    plt.title('Total Sales by Category')
    plt.ylabel('Total Sales (USD)')
    plt.show()
    return merged_data

from itertools import combinations
from collections import Counter

# Group products by CustomerID
customer_products = merged_data.groupby('CustomerID')['ProductID'].apply(list)

# Generate product pairs for each customer and count their occurrences
product_combinations = Counter([
    tuple(sorted(combo))
    for products in customer_products
    for combo in combinations(products, 2)
])

# Print the 10 most common product pairs
print("Top 10 Product Pairs:")
print(product_combinations.most_common(10))

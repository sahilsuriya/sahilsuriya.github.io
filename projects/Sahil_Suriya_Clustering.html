# Aggregate transactions per customer
customer_data = merged_data.groupby('CustomerID').agg(
    total_transactions=('TransactionID', 'nunique'),
    total_spending=('TotalValue', 'sum'),
    avg_spending_per_transaction=('TotalValue', 'mean'),
    total_quantity=('Quantity', 'sum'),
    unique_products_bought=('ProductID', 'nunique')
).reset_index()

import matplotlib.pyplot as plt
from sklearn.decomposition import PCA

# Apply PCA to reduce to 2D
pca = PCA(n_components=2)
pca_components = pca.fit_transform(scaled_features)

# Add PCA components to the data
customer_data['PCA1'] = pca_components[:, 0]
customer_data['PCA2'] = pca_components[:, 1]

# Get unique clusters
unique_clusters = customer_data['Cluster'].unique()

# Plot the clusters with labels
plt.figure(figsize=(8, 6))

for cluster in unique_clusters:
    cluster_data = customer_data[customer_data['Cluster'] == cluster]
    plt.scatter(cluster_data['PCA1'], cluster_data['PCA2'], label=f'Cluster {cluster}', alpha=0.7)

plt.title('Customer Segments')
plt.xlabel('PCA 1')
plt.ylabel('PCA 2')
plt.legend(title='Clusters')
plt.show()
